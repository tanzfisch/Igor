216  bool OMPF::analyze(ifstream& file)
217  {
218  if (!_root->read(file, _settings))
219  {
220  return false;
221  }
222 
223  OMPFChunkType typeID;
224  uint32 chunkSize;
225  uint32 tempType;
226  ompfBaseChunk* chunk = nullptr;
227  ompfMaterialChunk* materialChunk = nullptr;
228 
229  while (!file.eof())
230  {
231  chunk = nullptr;
232  materialChunk = nullptr;
233 
234  if (!getNextChunk(file, tempType, chunkSize))
235  {
236  return true;
237  }
238 
239  typeID = static_cast<OMPFChunkType>(tempType);
240  bool result = true;
241 
242  switch (typeID)
243  {
244  case OMPFChunkType::Header:
245  con_err("can't have second header chunk");
246  return false;
247 
248  case OMPFChunkType::Transform:
249  chunk = new ompfTransformChunk();
250  result = chunk->read(file, _settings);
251  break;
252 
253  case OMPFChunkType::External:
254  chunk = new ompfExternalReferenceChunk();
255  result = chunk->read(file, _settings);
256  break;
257 
258  case OMPFChunkType::Group:
259  chunk = new ompfGroupChunk();
260  result = chunk->read(file, _settings);
261  break;
262 
263  case OMPFChunkType::ResourceSearchPath:
264  chunk = new ompfResourceSearchPathChunk();
265  result = chunk->read(file, _settings);
266  break;
267 
268  case OMPFChunkType::Mesh:
269  chunk = new ompfMeshChunk();
270  result = chunk->read(file, _settings);
271  break;
272 
273  case OMPFChunkType::Emitter:
274  chunk = new ompfEmitterChunk();
275  result = chunk->read(file, _settings);
276  break;
277 
278  case OMPFChunkType::ParticleSystem:
279  chunk = new ompfParticleSystemChunk();
280  result = chunk->read(file, _settings);
281  break;
282 
283  case OMPFChunkType::Material:
284  chunk = new ompfMaterialChunk();
285  result = chunk->read(file, _settings);
286 
287  materialChunk = static_cast<ompfMaterialChunk*>(chunk);
288  _materialChunks.push_back(materialChunk);
289 
290  chunk = nullptr;
291  break;
292 
293  default:
294  file.seekg(chunkSize, file.cur);
295  con_warn("skipped unknown chunk with TypeID 0x" << hex << static_cast<int>(typeID) << " and size " << dec << chunkSize << " byte");
296  break;
297  }
298 
299  if (result)
300  {
301  if (chunk != nullptr)
302  {
303  _chunks[static_cast<uint32>(chunk->getID())] = chunk;
304  if (static_cast<uint32>(chunk->getID()) > _nextChunkID)
305  {
306  _nextChunkID = static_cast<uint32>(chunk->getID());
307  }
308  }
309  }
310  else
311  {
312  return false;
313  }
314  }
315 
316  return true;
317  }
