289  uint32 ompfParticleSystemChunk::getSize(const ompfSettings& settings)
290  {
291  uint32 result = 0;
292  result += 2; // max particle count
293  result += 1; // loop
294  result += static_cast<uint32>(_colorGradient.getValues().size()) * (sizeof(float32) * (1+4));
295  result += static_cast<uint32>(_emissionGradient.getValues().size()) * (sizeof(float32) * (1+1));
296  result += static_cast<uint32>(_sizeScaleGradient.getValues().size()) * (sizeof(float32) * 2);
297  result += static_cast<uint32>(_orientationGradient.getValues().size()) * (sizeof(float32) * (1+2));
298  result += static_cast<uint32>(_orientationRateGradient.getValues().size()) * (sizeof(float32) * (1+2));
299  result += static_cast<uint32>(_liftGradient.getValues().size()) * (sizeof(float32) * 3);
300  result += static_cast<uint32>(_velocityGradient.getValues().size()) * (sizeof(float32) * 3);
301  result += static_cast<uint32>(_sizeGradient.getValues().size()) * (sizeof(float32) * 3);
302  result += static_cast<uint32>(_startVisibleTimeGradient.getValues().size()) * (sizeof(float32) * 3);
303  result += 4 + 4; //min max vortex torque
304  result += 4 + 4; //min max vortex range
305  result += 1; // vortex check range
306  result += 4; // vortexToParticleRate
307  result += 4; // vorticity confinement
308  result += 2; // tiling column row
309  result += 4; // air drag
310  result += 4; // period time
311  result += 1; // velocity oriented
312  result += static_cast<uint32>(_textureA.getSize()) + 2;
313  result += static_cast<uint32>(_textureB.getSize()) + 2;
314  result += static_cast<uint32>(_textureC.getSize()) + 2;
315  result += 4; // material chunk id
316  result += 4; // emitter chunk id
317  return result; // 24
318  }
