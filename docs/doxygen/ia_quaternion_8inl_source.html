276 template <class T>
277 void iaQuaternion<T>::getEuler(T& x, T& y, T& z) const
278 {
279  // roll (x-axis rotation)
280  double sinr_cosp = +2.0 * (_w * _x + _y * _z);
281  double cosr_cosp = +1.0 - 2.0 * (_x * _x + _y * _y);
282  x = atan2(sinr_cosp, cosr_cosp);
283 
284  // pitch (y-axis rotation)
285  double sinp = +2.0 * (_w * _y - _z * _x);
286  if (fabs(sinp) >= 1)
287  y = copysign(M_PI / 2, sinp); // use 90 degrees if out of range
288  else
289  y = asin(sinp);
290 
291  // yaw (z-axis rotation)
292  double siny_cosp = +2.0 * (_w * _z + _x * _y);
293  double cosy_cosp = +1.0 - 2.0 * (_y * _y + _z * _z);
294  z = atan2(siny_cosp, cosy_cosp);
295 }
