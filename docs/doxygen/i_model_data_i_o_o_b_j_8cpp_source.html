126  void iModelDataIOOBJ::transferToMeshBuilder(Section & section)
127  {
128  auto polygons = section._polygons;
129  auto& meshBuilder = section._meshBuilder;
130 
131  for (auto polygon : polygons)
132  {
133  bool hasNormals = polygon._vertexes[0]._vn != -1 ? true : false;
134  bool hasTexCoord = polygon._vertexes[0]._t != -1 ? true : false;
135 
136  // asuming poly is a triangle fan
137  for (uint32 j = 0; j < polygon._vertexes.size() - 2; j++)
138  {
139  uint32 a = meshBuilder.addVertex(_vertexes[polygon._vertexes[0]._v]);
140  uint32 b = meshBuilder.addVertex(_vertexes[polygon._vertexes[j + 1]._v]);
141  uint32 c = meshBuilder.addVertex(_vertexes[polygon._vertexes[j + 2]._v]);
142 
143  if (hasNormals)
144  {
145  meshBuilder.setNormal(a, _normals[polygon._vertexes[0]._vn]);
146  meshBuilder.setNormal(b, _normals[polygon._vertexes[j + 1]._vn]);
147  meshBuilder.setNormal(c, _normals[polygon._vertexes[j + 2]._vn]);
148  }
149 
150  if (hasTexCoord)
151  {
152  meshBuilder.setTexCoord(a, _texcoord[polygon._vertexes[0]._t], 0);
153  meshBuilder.setTexCoord(b, _texcoord[polygon._vertexes[j + 1]._t], 0);
154  meshBuilder.setTexCoord(c, _texcoord[polygon._vertexes[j + 2]._t], 0);
155  }
156 
157  meshBuilder.addTriangle(a, b, c);
158  }
159  }
