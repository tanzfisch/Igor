374 template <class T>
375 __IGOR_INLINE__ bool iaMatrix<T>::invert()
376 {
377  T* mat = (T*)& _right;
378  T matr[4][4], ident[4][4];
379  int i, j, k, l, ll;
380  int icol = 0, irow = 0;
381  int indxc[4], indxr[4], ipiv[4];
382  T big, dum, pivinv, temp;
383 
384  for (i = 0; i < 4; i++)
385  {
386  for (j = 0; j < 4; j++)
387  {
388  matr[i][j] = mat[4 * i + j];
389  ident[i][j] = 0.0f;
390  }
391  ident[i][i] = 1.0f;
392  }
393 
394  // Gauss-jordan elimination with full pivoting. Yes, folks, a
395  // GL Matrix4 is inverted like any other, since the identity is
396  // still the identity.
397 
398  // from numerical recipies in C second edition, pg 39
399 
400  for (j = 0; j <= 3; j++)
401  {
402  ipiv[j] = 0;
403  }
404 
405  for (i = 0; i <= 3; i++)
406  {
407  big = 0.0;
408  for (j = 0; j <= 3; j++)
409  {
410  if (ipiv[j] != 1)
411  {
412  for (k = 0; k <= 3; k++)
413  {
414  if (ipiv[k] == 0)
415  {
416  if (fabs(matr[j][k]) >= big)
417  {
418  big = (T)fabs(matr[j][k]);
419  irow = j;
420  icol = k;
421  }
422  }
423  else if (ipiv[k] > 1)
424  {
425  return false;
426  }
427  }
428  }
429  }
430 
431  ++(ipiv[icol]);
432  if (irow != icol)
433  {
434  for (l = 0; l <= 3; l++)
435  {
436  MATSWAP(matr[irow][l], matr[icol][l]);
437  }
438 
439  for (l = 0; l <= 3; l++)
440  {
441  MATSWAP(ident[irow][l], ident[icol][l]);
442  }
443  }
444 
445  indxr[i] = irow;
446  indxc[i] = icol;
447  if (matr[icol][icol] == 0.0f)
448  {
449  return false;
450  }
451 
452  pivinv = 1.0f / matr[icol][icol];
453  matr[icol][icol] = 1.0f;
454  for (l = 0; l <= 3; l++)
455  {
456  matr[icol][l] *= pivinv;
457  }
458 
459  for (l = 0; l <= 3; l++)
460  {
461  ident[icol][l] *= pivinv;
462  }
463 
464  for (ll = 0; ll <= 3; ll++)
465  {
466  if (ll != icol)
467  {
468  dum = matr[ll][icol];
469  matr[ll][icol] = 0.0f;
470  for (l = 0; l <= 3; l++)
471  {
472  matr[ll][l] -= matr[icol][l] * dum;
473  }
474 
475  for (l = 0; l <= 3; l++)
476  {
477  ident[ll][l] -= ident[icol][l] * dum;
478  }
479  }
480  }
481  }
482 
483  for (l = 3; l >= 0; l--)
484  {
485  if (indxr[l] != indxc[l])
486  {
487  for (k = 0; k <= 3; k++)
488  {
489  MATSWAP(matr[k][indxr[l]], matr[k][indxc[l]]);
490  }
491  }
492  }
493 
494  for (i = 0; i < 4; i++)
495  {
496  for (j = 0; j < 4; j++)
497  {
498  mat[4 * i + j] = matr[i][j];
499  }
500  }
501  return true;
502 }

740 template <class T>
741 bool iaMatrix<T>::decompose(iaVector3<T> & scale, iaQuaternion<T> & orientation, iaVector3<T> & translate, iaVector3<T> & shear, iaVector4<T> & perspective)
742 {
743  iaMatrix<T> localMatrix(*this);
744 
745  // Normalize the matrix.
746  if (_w3 == 0)
747  {
748  return false;
749  }
750 
751  for (int i = 0; i < 16; ++i)
752  {
753  localMatrix[i] /= localMatrix._w3;
754  }
755 
756  // perspectiveMatrix is used to solve for perspective, but it also provides
757  // an easy way to test for singularity of the upper 3x3 component.
758  iaMatrix<T> perspectiveMatrix(localMatrix);
759  perspectiveMatrix._w0 = 0;
760  perspectiveMatrix._w1 = 0;
761  perspectiveMatrix._w2 = 0;
762  perspectiveMatrix._w3 = 1;
763 
764  if (perspectiveMatrix.determinant() == 0)
765  {
766  return false;
767  }
768 
769  // First, isolate perspective. This is the messiest.
770  if (localMatrix._w0 != 0 || localMatrix._w1 != 0 || localMatrix._w2 != 0)
771  {
772  // rightHandSide is the right hand side of the equation.
773  iaVector4<T> rightHandSide;
774  rightHandSide[0] = localMatrix._w0;
775  rightHandSide[1] = localMatrix._w1;
776  rightHandSide[2] = localMatrix._w2;
777  rightHandSide[3] = localMatrix._w3;
778 
779  // Solve the equation by inverting perspectiveMatrix and multiplying
780  // rightHandSide by the inverse. (This is the easiest way, not
781  // necessarily the best.)
782  iaMatrix<T> transposedInversePerspectiveMatrix(perspectiveMatrix);
783  if (!transposedInversePerspectiveMatrix.invert())
784  {
785  return false;
786  }
787  transposedInversePerspectiveMatrix.transpose();
788 
789  perspective = transposedInversePerspectiveMatrix * rightHandSide;
790 
791  // Clear the perspective partition
792  localMatrix._w0 = 0;
793  localMatrix._w1 = 0;
794  localMatrix._w2 = 0;
795  localMatrix._w3 = 1;
796  }
797  else
798  {
799  // No perspective.
800  perspective.set(0, 0, 0, 1);
801  }
802 
803  // Next take care of translation (easy).
804  translate = localMatrix._pos;
805  localMatrix._pos.set(0, 0, 0);
806 
807  // Vector4 type and functions need to be added to the common set.
808  iaVector3<T> row[3], pdum3;
809 
810  // Now get scale and shear.
811  row[0].set(localMatrix[0], localMatrix[4], localMatrix[8]);
812  row[1].set(localMatrix[1], localMatrix[5], localMatrix[9]);
813  row[2].set(localMatrix[2], localMatrix[6], localMatrix[10]);
814 
815  // Compute X scale factor and normalize first row.
816  scale._x = row[0].length();
817  row[0].normalize();
818 
819  // Compute XY shear factor and make 2nd row orthogonal to 1st.
820  shear._z = row[0] * row[1];
821  row[1] = row[1] + (row[0] * -shear._z);
822 
823  // Now, compute Y scale and normalize 2nd row.
824  scale._y = row[1].length();
825  row[1].normalize();
826  shear._z /= scale._y;
827 
828  // Compute XZ and YZ shears, orthogonalize 3rd row.
829  shear._y = row[0] * row[2];
830  row[2] = row[2] + (row[0] * -shear._y);
831  shear._x = row[1] * row[2];
832  row[2] = row[2] + (row[1] * -shear._x);
833 
834  // Next, get Z scale and normalize 3rd row.
835  scale._z = row[2].length();
836  row[2].normalize();
837  shear._y /= scale._z;
838  shear._x /= scale._z;
839 
840  // At this point, the matrix (in rows[]) is orthonormal.
841  // Check for a coordinate system flip. If the determinant
842  // is -1, then negate the matrix and the scaling factors.
843  pdum3 = row[1] % row[2];
844  if ((row[0] * pdum3) < 0)
845  {
846  scale._x *= -1;
847  row[0].negate();
848  row[1].negate();
849  row[2].negate();
850  }
851 
852  // Now, get the rotations out
853  T s, t;
854 
855  t = row[0][0] + row[1][1] + row[2][2] + 1.0;
856 
857  if (t > 1e-4) {
858  s = 0.5 / sqrt(t);
859  orientation._w = 0.25 / s;
860  orientation._x = (row[2][1] - row[1][2]) * s;
861  orientation._y = (row[0][2] - row[2][0]) * s;
862  orientation._z = (row[1][0] - row[0][1]) * s;
863  }
864  else if (row[0][0] > row[1][1] && row[0][0] > row[2][2]) {
865  s = sqrt(1.0 + row[0][0] - row[1][1] - row[2][2]) * 2.0; // S=4*qx
866  orientation._x = 0.25 * s;
867  orientation._y = (row[0][1] + row[1][0]) / s;
868  orientation._z = (row[0][2] + row[2][0]) / s;
869  orientation._w = (row[2][1] - row[1][2]) / s;
870  }
871  else if (row[1][1] > row[2][2]) {
872  s = sqrt(1.0 + row[1][1] - row[0][0] - row[2][2]) * 2.0; // S=4*qy
873  orientation._x = (row[0][1] + row[1][0]) / s;
874  orientation._y = 0.25 * s;
875  orientation._z = (row[1][2] + row[2][1]) / s;
876  orientation._w = (row[0][2] - row[2][0]) / s;
877  }
878  else {
879  s = sqrt(1.0 + row[2][2] - row[0][0] - row[1][1]) * 2.0; // S=4*qz
880  orientation._x = (row[0][2] + row[2][0]) / s;
881  orientation._y = (row[1][2] + row[2][1]) / s;
882  orientation._z = 0.25 * s;
883  orientation._w = (row[1][0] - row[0][1]) / s;
884  }
885 
886  return true;
887 }
