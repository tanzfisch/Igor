55  _windowsFileHandle = CreateFile(getFullFileName().getData(), accessMode, shareMode, 0, openMode, FILE_ATTRIBUTE_NORMAL, 0);
56  if(_windowsFileHandle == INVALID_HANDLE_VALUE)
57  {
58  con_err_win("cannot open/create file: " << getFullFileName());
59  _windowsFileHandle = 0;
60  return false;
61  }
62 
63  // get the 64bit size
64  DWORD sizeHigh = 0;
65  const DWORD sizeLow = GetFileSize(_windowsFileHandle, &sizeHigh);
66  if(sizeLow == INVALID_FILE_SIZE)
67  {

69  const DWORD error = GetLastError();
70  if(error != NO_ERROR)
71  {
72  con_err_win("cannot get file size: " << getFullFileName());
73  return false;
74  }
75  }
