400 {
401  if (node != nullptr)

405  iSphered sphere;
406  visitorBoundings.getSphere(sphere);
407 
408  iaMatrixd coiMatrix;

411  _manipulator->setCamCOI(coiMatrix);
412

414  {
415  _camDistance = sphere._radius * 4.0f;
416  }

418  {
419  _camDistance = 1.0f;

864 void ModelViewer::onMouseMoved(const iaVector2i& from, const iaVector2i& to, iWindow* window)
865 {
866  const float64 rotateSensitivity = 0.0075;
867  const float64 translateSensitivity = 1.0;
868 
869  if (iMouse::getInstance().getLeftButton())
870  {
871  if (iKeyboard::getInstance().getKey(iKeyCode::LAlt))
872  {
873  _cameraPitch->rotate((from._y - to._y) * rotateSensitivity, iaAxis::X);
874  _cameraHeading->rotate((from._x - to._x) * rotateSensitivity, iaAxis::Y);
875 
876  iaMatrixd matrix;
877  _cameraPitch->getMatrix(matrix);
878  _manipulator->setCamPitch(matrix);
879 
880  _cameraHeading->getMatrix(matrix);
881  _manipulator->setCamHeading(matrix);
882  }
883  else
884  {
885  _manipulator->onMouseMoved(from, to, window);
886  }
887  }
888 
889  if (iMouse::getInstance().getRightButton())
890  {
891  _directionalLightRotate->rotate((from._y - to._y) * rotateSensitivity, iaAxis::X);
892  _directionalLightRotate->rotate((from._x - to._x) * rotateSensitivity, iaAxis::Y);
893  }
894 
895  if (iMouse::getInstance().getMiddleButton())
896  {
897  if (iKeyboard::getInstance().getKey(iKeyCode::LAlt))
898  {
899  iaMatrixd camWorldMatrix;
900  _camera->calcWorldTransformation(camWorldMatrix);
901  iaVector3d fromWorld = camWorldMatrix * _view.unProject(iaVector3d(from._x, from._y, 0), camWorldMatrix);
902  iaVector3d toWorld = camWorldMatrix * _view.unProject(iaVector3d(to._x, to._y, 0), camWorldMatrix);
903 
904  iaMatrixd camTranslateMatrix;
905  _cameraTranslation->getMatrix(camTranslateMatrix);
906  float64 translateFactor = camTranslateMatrix._pos.length() * translateSensitivity;
907 
908  _cameraCOI->translate((fromWorld - toWorld) * translateFactor);
909  }
910  }
911 }
