392  __IGOR_INLINE__ iaConsole& generateWindowsError(iaConsole &console)
393  {
394  int error = GetLastError();
395 
396  if (error != 0)
397  {
398  char * errorString = 0;
399  FormatMessage(FORMAT_MESSAGE_ALLOCATE_BUFFER | FORMAT_MESSAGE_FROM_SYSTEM,
400  0, error, MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT), (LPTSTR)& errorString, 0, 0);
401 
402  if (errorString)
403  {
404  console << errorString;
405  if (console._streamToLogfile && console._file.is_open())
406  {
407  console._file << errorString;
408  }
409  LocalFree(errorString);
410  }
411  }
412  else
413  {
414  console << "no error code";
415  if (console._streamToLogfile && console._file.is_open())
416  {
417  console._file << "no error code";
418  }
419  }
420 
421  return console;
422  }

425  {

433  console << setfill(L'0') << setw(3) << time % 1000 << " ";
