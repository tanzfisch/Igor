345 void UserControlGraphView::setSelectedNode(uint64 nodeID)
346 {
347  if (nodeID == iNode::INVALID_NODE_ID)
348  {
349  _gridGraph->unSelect();
350  return;
351  }
352 
353  int32 rowCount = _gridGraph->getRowCount();
354  for (int row = 0; row < rowCount; ++row)
355  {
356  uint64* nodeIDPtr = static_cast<uint64*>(_gridGraph->getUserData(0, row));
357  uint64 id = iNode::INVALID_NODE_ID;
358  if (nodeIDPtr != nullptr)
359  {
360  id = *nodeIDPtr;
361  }
362 
363  if (nodeID == id)
364  {
365  _gridGraph->select(0, row);
366  return;
367  }
368  }
369 
370  _gridGraph->unSelect();
371 }
