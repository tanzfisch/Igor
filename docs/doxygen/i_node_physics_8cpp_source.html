322  uint64 iNodePhysics::initPhysics()
323  {
324  uint64 result = iPhysicsBody::INVALID_PHYSICSBODY_ID;
325 
326  if (!_physicsInitialized)
327  {
328  iPhysicsCollisionConfig* physicsCollisionConfig = iPhysics::getInstance().getCollisionConfig(_physicsCollisionConfigID);
329  if (physicsCollisionConfig && physicsCollisionConfig->isFinalized())
330  {
331  iPhysicsCollision* collision = iPhysics::getInstance().getCollision(physicsCollisionConfig->getCollisionID());
332  iPhysicsBody* body = iPhysics::getInstance().createBody(collision);
333  body->setMass(_mass);
334  body->setMaterial(_materialID);
335  body->setUserData(getUserData());
336  body->setLinearDamping(_linearDamping);
337  body->setAngularDamping(_angularDamping);
338  body->setForce(_force);
339  body->setTorque(_torque);
340 
341  if (_applyForceAndTorqueDelegate != nullptr)
342  {
343  body->registerForceAndTorqueDelegate(*_applyForceAndTorqueDelegate);
344  delete _applyForceAndTorqueDelegate;
345  _applyForceAndTorqueDelegate = nullptr;
346  }
347 
348  result = body->getID();
349 
350  iPhysics::getInstance().destroyCollision(collision);
351  iPhysics::getInstance().destroyCollisionConfig(_physicsCollisionConfigID);
352  _physicsCollisionConfigID = iPhysicsCollisionConfig::INVALID_COLLISIONCONFIG_ID;
353

417  bool iNodePhysics::onUpdateData()
418  {
419  iNodeTransform* nextTransformNode = static_cast<iNodeTransform*>(_parent);
420  while (nextTransformNode != nullptr && nextTransformNode->getType() != iNodeType::iNodeTransform)
421  {
422  nextTransformNode = static_cast<iNodeTransform*>(nextTransformNode->getParent());
423  }
424 
425  if (nextTransformNode != nullptr &&
426  nextTransformNode->getType() == iNodeType::iNodeTransform)
427  {
428  _bodyID = initPhysics();
429 
430  if (_bodyID != iPhysicsBody::INVALID_PHYSICSBODY_ID)
431  {
432  iPhysicsBody* body = iPhysics::getInstance().getBody(_bodyID);
433  iPhysics::getInstance().bindTransformNode(body, nextTransformNode, true);
434  }
435  else
436  {
437  // physics is not ready yet
438  return false;
439  }
440  }
441  else
442  {
443  con_warn("need a transform node as ancester");
444  return false;
445  }
