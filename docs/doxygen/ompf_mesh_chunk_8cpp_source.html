179  uint32 ompfMeshChunk::getSize(const ompfSettings & settings)
180  {
181  uint32 material = 3; // ambient
182  material += 3; // diffuse
183  material += 3; // specular
184  material += 3; // emissive
185  material += 4; // shininess
186  material += 4; // alpha
187  uint32 attributes = 1; // normals per vertex
188  attributes += 1; // colors per vertex
189  attributes += 1; // texcoord per vertex
190  attributes += 1; // mesh type
191  attributes += 4; // vertex count
192  attributes += 4; // index count
193  attributes += 4; // material chunk id
194  uint32 textures = 0;
195 
196  auto iterTex = _textures.begin();
197  while (iterTex != _textures.end())
198  {
199  textures += static_cast<uint32>((*iterTex).second.getSize()) + 2;
200  iterTex++;
201  }
202 
203  return material + attributes + _vertexDataSize + _indexDataSize + textures;
204  }
