134  void iNodeFactory::flush()
135  {
136  _mutexQueue.lock();
137  auto queue = std::move(_actionQueue);
138  _mutexQueue.unlock();
139 
140  iNodePtr nodeA = nullptr;
141  iNodePtr nodeB = nullptr;
142 
143  for (auto entry : queue)
144  {
145  switch (entry._action)
146  {
147  case iActionType::Insert:
148  nodeA = getNode(entry._nodeA);
149  nodeB = getNode(entry._nodeB);
150  if (nodeA != nullptr && nodeB != nullptr)
151  {
152  nodeA->insertNode(nodeB);
153  }
154  break;
155 
156  case iActionType::Remove:
157  nodeA = getNode(entry._nodeA);
158  nodeB = getNode(entry._nodeB);
159  if (nodeA != nullptr && nodeB != nullptr)
160  {
161  nodeA->removeNode(nodeB);
162  }
163  break;
164 
165  case iActionType::Destroy:
166  destroyNode(entry._nodeA);
167  break;
168 
169  case iActionType::Activate:
170  nodeA = getNode(entry._nodeA);
171  if (nodeA != nullptr)
172  {
173  nodeA->setActive(true);
174  }
175  break;
176 
177  case iActionType::Deactivate:
178  nodeA = getNode(entry._nodeA);
179  if (nodeA != nullptr)
180  {
181  nodeA->setActive(false);
182  }
183  break;
184  }
