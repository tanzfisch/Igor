424  void iSimpleCubes::compile(iMeshBuilder& meshBuilder, iaVector3I pos, iaVector3I volume)
425  {
426  if (pos._x < 0 ||
427  pos._y < 0 ||
428  pos._z < 0)
429  {
430  con_err("out of range");
431  return;
432  }
433 
434  iaVector3I max = pos;
435  max += volume;
436  if (max._x >= _voxelData->getWidth() ||
437  max._y >= _voxelData->getDepth() ||
438  max._z >= _voxelData->getHeight())
439  {
440  con_err("out of range");
441  return;
442  }
443 
444  _marchingVolume = volume;
445 
446  _cubeStartPosition.set(pos._x, pos._y, pos._z);
447 
448  iaVector3I currentPosition;
449  currentPosition._z = _cubeStartPosition._z;
450 
451  // run through collumns
452  for (int x = 0; x < _marchingVolume._x - 1; ++x)
453  {
454  currentPosition._x = _cubeStartPosition._x + x;
455 
456  // run through rows
457  for (int y = 0; y < _marchingVolume._y - 1; ++y)
458  {
459  currentPosition._y = _cubeStartPosition._y + y;
460 
461  // process pole
462  startGettingCubes(currentPosition);
463  do
464  {
465  generateGeometry(meshBuilder, getNextCube());
466  } while (!(_cubePosition._z >= _marchingVolume._z + _cubeStartPosition._z - 1));
467  }
468  }
469 
470  if (meshBuilder.getTrianglesCount() != 0)
471  {
472  meshBuilder.calcNormals(false);
473  }
474  }
