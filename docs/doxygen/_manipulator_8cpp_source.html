301 void Manipulator::update()
302 {
303  iaMatrixd matrix;
304 
305  iNodePtr node = iNodeFactory::getInstance().getNode(_selectedNodeID);
306  if (node != nullptr)
307  {
308  node->calcWorldTransformation(matrix);
309  }
310  else
311  {
312  setVisible(false);
313  return;
314  }
315 
316  iaMatrixd camMatrix;
317  _cameraUI->calcWorldTransformation(camMatrix);
318 
319  float64 distanceToCam = camMatrix._pos.distance(matrix._pos) * 0.1;
320 
321  matrix._right.normalize();
322  matrix._top.normalize();
323  matrix._depth.normalize();
324  _rootTransform->setMatrix(matrix);
325  _rootTransform->scale(distanceToCam, distanceToCam, distanceToCam);
326 
327  // compensate for parent orientation
328  iaMatrixd parentMatrix;
329  _rotateBillboardTransform->getParent()->calcWorldTransformation(parentMatrix);
330  parentMatrix._pos.set(0,0,0);
331  parentMatrix.invert();
332  parentMatrix._right.normalize();
333  parentMatrix._top.normalize();
334  parentMatrix._depth.normalize();
335 
336  _rotateBillboardTransform->identity();
337  iaMatrixd billboardMatrix;
338  _rotateBillboardTransform->getMatrix(billboardMatrix);
339  billboardMatrix._right = camMatrix._right;
340  billboardMatrix._top = camMatrix._top;
341  billboardMatrix._depth = camMatrix._depth;
342  _rotateBillboardTransform->setMatrix(parentMatrix * billboardMatrix);
343  _rotateBillboardTransform->rotate(M_PI * 0.5, iaAxis::X);
344  _rotateBillboardTransform->scale(2.1, 2.1, 2.1);
345 }

540 void Manipulator::rotate(const iaVector2d& from, const iaVector2d& to, iaMatrixd& matrix)
541 {
542  iNode* node = iNodeFactory::getInstance().getNode(_selectedNodeID);
543  iaMatrixd transformWorldMatrix;
544  node->calcWorldTransformation(transformWorldMatrix);
545 
546  iaMatrixd camWorldMatrix;
547  _cameraUI->calcWorldTransformation(camWorldMatrix);
548  iaVector3d center = _view->project(transformWorldMatrix._pos, camWorldMatrix);
549 
550  iaVector2d center2D(center._x, center._y);
551 
552  iaVector2d a = from - center2D;
553  iaVector2d b = to - center2D;
554 
555  float64 angle = b.angle() - a.angle();
556 
557  for (int i = 0; i < 3; ++i)
558  {
559  if (_selectedManipulatorNodeID == _rotateIDs[i])
560  {
561  iaAxis axis = static_cast<iaAxis>(i);
562  float64 scalar = 0;
563 
564  iaVector3d toCam = camWorldMatrix._pos - matrix._pos;
565 
566  switch (axis)
567  {
568  case iaAxis::X:
569  scalar = toCam * matrix._right;
570  break;
571  case iaAxis::Y:
572  scalar = toCam * matrix._top;
573  break;
574  case iaAxis::Z:
575  scalar = toCam * matrix._depth;
576  break;
577  }
578 
579  if (scalar < 0)
580  {
581  angle = -angle;
582  }
583 
584  matrix.rotate(angle, static_cast<iaAxis>(i));
585  return;
586  }
587  }
