76 void UserControlMesh::updateNode()
77 {
78  if (!_ignoreNodeUpdate)
79  {
80  iNodeMesh* node = static_cast<iNodeMesh*>(iNodeFactory::getInstance().getNode(_nodeId));
81 
82  if (node != nullptr)
83  {
84  iaColor3f ambient(_ambient._r, _ambient._g, _ambient._b);
85  iaColor3f diffuse(_diffuse._r, _diffuse._g, _diffuse._b);
86  iaColor3f specular(_specular._r, _specular._g, _specular._b);
87  iaColor3f emissive(_emissive._r, _emissive._g, _emissive._b);
88 
89  node->getTargetMaterial()->setAmbient(ambient);
90  node->getTargetMaterial()->setDiffuse(diffuse);
91  node->getTargetMaterial()->setSpecular(specular);
92  node->getTargetMaterial()->setEmissive(emissive);
93  node->getTargetMaterial()->setShininess(_shininess);
94 
95  node->getTargetMaterial()->setTexture(iTextureResourceFactory::getInstance().loadFile(_textureChooser0->getFileName()), 0);
96  node->getTargetMaterial()->setTexture(iTextureResourceFactory::getInstance().loadFile(_textureChooser1->getFileName()), 1);
97  node->getTargetMaterial()->setTexture(iTextureResourceFactory::getInstance().loadFile(_textureChooser2->getFileName()), 2);
98  node->getTargetMaterial()->setTexture(iTextureResourceFactory::getInstance().loadFile(_textureChooser3->getFileName()), 3);
99 
100  if (_selectMaterial->getSelectedUserData() != nullptr)
101  {
102  uint32 materialID = *(static_cast<uint32*>(_selectMaterial->getSelectedUserData()));
103  node->setMaterial(materialID);
104  }
105  }
106  }
