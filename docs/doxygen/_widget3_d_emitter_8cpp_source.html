75 void Widget3DEmitter::update()
76 {
77  iNodePtr node = iNodeFactory::getInstance().getNode(_nodeID);
78  if (node == nullptr)

80  clearMeshNode();
81  return;
82  }
83 
84  if (node->getType() != iNodeType::iNodeEmitter)
85  {
86  clearMeshNode();
87  return;
88  }
89 
90  // create structure
91  if (_rootTransform == nullptr)

93  _rootTransform = static_cast<iNodeTransform*>(iNodeFactory::getInstance().createNode(iNodeType::iNodeTransform));
94  _scene->getRoot()->insertNode(_rootTransform);
95 
96  _scaleTransform = static_cast<iNodeTransform*>(iNodeFactory::getInstance().createNode(iNodeType::iNodeTransform));
97  _rootTransform->insertNode(_scaleTransform);
98  }
99 
100  iNodeEmitter* emitter = static_cast<iNodeEmitter*>(node);
101  iEmitterType emitterType = emitter->getEmitterType();
102 
103  // change emitter type
104  if (_emitterType != emitterType)
105  {
106  _emitterType = emitterType;
107  clearMeshNode();
108 
109  // create mesh
110  shared_ptr<iMesh> mesh;
111  uint64 material = _material;
112  switch (_emitterType)
113  {
114  case iEmitterType::Mesh:
115  // not supported yet
116  break;
117 
118  case iEmitterType::Point:
119  // show nothing
120  break;
121 
122  case iEmitterType::Disc:
123  mesh = createDisc();
124  break;
125 
126  case iEmitterType::Circle:
127  mesh = createCircle();
128  break;
129 
130  case iEmitterType::Sphere:
131  mesh = createSphere();
132  material = _materialVolume;
133  break;
134 
135  case iEmitterType::Square:
136  mesh = createSquare();
137  break;
138 
139  case iEmitterType::Cube:
140  mesh = createCube();
141  material = _materialVolume;
142  break;
143  }
144 
145  if (mesh != nullptr)
146  {
147  _meshNode = static_cast<iNodeMesh*>(iNodeFactory::getInstance().createNode(iNodeType::iNodeMesh));
148  _meshNode->setMesh(mesh);
149  _meshNode->setMaterial(material);
150  _meshNode->setTargetMaterial(_targetMaterial);
151  _scaleTransform->insertNode(_meshNode);
152  }
153  }
154 
155  // update size
156  _scaleTransform->identity();
157  _scaleTransform->scale(emitter->getSize(), emitter->getSize(), emitter->getSize());
158 
159  // update matrix

161  node->calcWorldTransformation(matrix);
162  _rootTransform->setMatrix(matrix);
163 }
