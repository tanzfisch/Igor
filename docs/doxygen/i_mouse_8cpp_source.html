80  bool iMouse::onOSEvent(const void *data)
81  {
82  con_assert(nullptr != _window, "not registered to a window");
83 
84  iOSEventWin32 *event = (iOSEventWin32*)data;
85 
86  switch (event->_uMsg)
87  {
88  case WM_INPUT:
89  {
90  if (!_rawInputBufferSize)
91  {
92  GetRawInputData((HRAWINPUT)event->_lParam, RID_INPUT, 0, &_rawInputBufferSize, sizeof(RAWINPUTHEADER));
93  _rawInputBuffer = new uint8[_rawInputBufferSize];
94  }
95 
96  GetRawInputData((HRAWINPUT)event->_lParam, RID_INPUT, (LPVOID)_rawInputBuffer, &_rawInputBufferSize, sizeof(RAWINPUTHEADER));
97 
98  RAWINPUT *raw = (RAWINPUT*)_rawInputBuffer;
99  if (raw->header.dwType == RIM_TYPEMOUSE)
100  {
101  if (raw->data.mouse.usButtonFlags & RI_MOUSE_BUTTON_4_DOWN)
102  {
103  _button4 = true;
104  _keyDownEvent(iKeyCode::MouseButton4);
105  }
106 
107  if (raw->data.mouse.usButtonFlags & RI_MOUSE_BUTTON_5_DOWN)
108  {
109  _button5 = true;
110  _keyDownEvent(iKeyCode::MouseButton5);
111  }
112 
113  if (raw->data.mouse.usButtonFlags & RI_MOUSE_BUTTON_4_UP)
114  {
115  _button4 = false;
116  _keyUpEvent(iKeyCode::MouseButton4);
117  }
118 
119  if (raw->data.mouse.usButtonFlags & RI_MOUSE_BUTTON_5_UP)
120  {
121  _button5 = false;
122  _keyUpEvent(iKeyCode::MouseButton5);
123  }
124  }
125  }
126  break;
127 
128  case WM_MOUSEMOVE:
129  if (_ignoreNextMoveEvent)
130  {
131  _ignoreNextMoveEvent = false;
132  }
133  else
134  {
135  _posLast = _pos;
136  _pos.set(GET_X_LPARAM(event->_lParam), GET_Y_LPARAM(event->_lParam));
137 
138  _moveFullEvent(_posLast, _pos, _window);
139  _moveEvent(_pos);
140  }
141  break;
142 
143  case WM_RBUTTONDOWN:
144  _rButton = true;
145  _keyDownEvent(iKeyCode::MouseRight);
146  break;
147 
148  case WM_RBUTTONUP:
149  if (_rButton)
150  {
151  _clickEvent(iKeyCode::MouseRight);
152  }
153 
154  _rButton = false;
155  _keyUpEvent(iKeyCode::MouseRight);
156  break;
157 
158  case WM_RBUTTONDBLCLK:
159  _doubleClickEvent(iKeyCode::MouseRight);
160  break;
161 
162  case WM_LBUTTONDOWN:
163  _lButton = true;
164  _keyDownEvent(iKeyCode::MouseLeft);
165  break;
166 
167  case WM_LBUTTONUP:
168  if (_lButton)
169  {
170  _clickEvent(iKeyCode::MouseLeft);
171  }
172 
173  _lButton = false;
174  _keyUpEvent(iKeyCode::MouseLeft);
175  break;
176 
177  case WM_LBUTTONDBLCLK:
178  _doubleClickEvent(iKeyCode::MouseLeft);
179  break;
180 
181  case WM_MBUTTONDOWN:
182  _mButton = true;
183  _keyDownEvent(iKeyCode::MouseMiddle);
184  break;
185 
186  case WM_MBUTTONUP:
187  if (_mButton)
188  {
189  _clickEvent(iKeyCode::MouseMiddle);
190  }
191 
192  _mButton = false;
193  _keyUpEvent(iKeyCode::MouseMiddle);
194  break;
195 
196  case WM_MBUTTONDBLCLK:
197  _doubleClickEvent(iKeyCode::MouseMiddle);
198  break;
199 
200  case WM_MOUSEWHEEL:
201  _wheelDelta = int32(int16(HIWORD(event->_wParam))) / int32(WHEEL_DELTA);
202  _wheelEvent(_wheelDelta);
203  break;
204 
205  default: return false;
206  }

216  void iMouse::setPosition(int32 x, int32 y, bool supressMoveEvent)
217  {
218  _ignoreNextMoveEvent = supressMoveEvent;
219 
220  if (_window->isFullscreen())
221  {
222  SetCursorPos(x, y);
223  }
224  else
225  {
226  // SetCursorPos seems to subtract the border so we have to add it twice
227  int32 borderX = (GetSystemMetrics(SM_CXSIZEFRAME) << 1);
228  int32 borderY = (GetSystemMetrics(SM_CYSIZEFRAME) << 1);
229  int32 headerY = GetSystemMetrics(SM_CYCAPTION);
230  int32 client_rect_offset_pos_x = borderX + _window->getXPos();
231  int32 client_rect_offset_pos_y = borderY + headerY + _window->getYPos();
232  SetCursorPos(client_rect_offset_pos_x + x, client_rect_offset_pos_y + y);
233  }
234 
235  _posLast = _pos;
236  _pos.set(x, y);
237  }
