112  (delta._z > 0) ? step._z = 1 : step._z = -1;
113 
114  if (delta._x < 0) delta._x = -delta._x;
115  if (delta._y < 0) delta._y = -delta._y;
116  if (delta._z < 0) delta._z = -delta._z;

119  dist = (dist > delta._z) ? dist : delta._z;
