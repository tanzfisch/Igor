94  {
95  if (y % 16 == 0)
96  {
97  if (black) black = false;
98  else black = true;

100

102  {
103  if (x % 16 == 0)
104  {
105  if (black) black = false;
106  else black = true;
107  }
108

110  {
111  data[(y*width + x)*Bpp + 0] = 0;
112  data[(y*width + x)*Bpp + 1] = 0;

120  data[(y*width + x)*Bpp + 2] = (uint8)(255 * (y / (float)height));
121  data[(y*width + x)*Bpp + 3] = 255;
122  }
123  }
124  }
