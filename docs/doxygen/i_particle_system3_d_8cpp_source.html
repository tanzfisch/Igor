420  {
421  if (_mustReset)
422  {
423  reset();
424  }
425 
426  if (_running)
427  {
428  iaVector3f a, b;
429 
430  uint32 startIndex;
431  uint32 endIndex;
432 
433  float64 frameTime = iTimer::getInstance().getMilliSeconds();
434 
435  // ignore hickups
436  if (frameTime - _playbackTime > 100.0)
437  {
438  _playbackTime = frameTime;
439  }
440 
441  float64 particleSystemTime = _playbackTime - _startTime;
442 
443  if (particleSystemTime >= _particleSystemPeriodTime)
444  {
445  if (_loop)
446  {

448  }
449  else

452  _running = false;
453  }
454  }

464  {

466 
467  particle._life -= 1.0f / _simulationRate;
468  if (particle._life <= 0)

471  }
472  else
473  {
474  _sizeScaleGradient.getValue(particle._visibleTime, sizeScale);
475 
476  particle._velocity[1] += particle._lift;
477 
478  particle._velocity *= _airDrag;
479 
480  particle._orientation += particle._orientationRate;
481 
482  particle._sizeScale = sizeScale;
483 
484  particle._phase0.rotateXY(_octave1Rotation);
485  particle._phase1.rotateXY(_octave2Rotation);
486 
487  particle._visibleTime += particle._visibleTimeIncrease;
488  if (particle._visibleTime > 1.0f)
489  {
490  particle._visible = false;
491  }
492 
493  if (particle._torque != 0.0)
494  {
495  _torqueFactorGradient.getValue(particle._visibleTime, torqueFactor);
496 
497  startIndex = index - _vortexCheckRange;
498  if (startIndex > 0)
499  {
500  startIndex = 0;
501  }
502 
503  endIndex = index + _vortexCheckRange;
504  if (endIndex > _particles.size())
505  {
506  endIndex = _particles.size();
507  }
508 
509  // TODO need overflow here
510 
511  for (uint32 i = startIndex; i < endIndex; ++i)
512  {
513  if (index == i) // ignore your self
514  {
515  continue;
516  }
517 
518  a = particle._position - _particles[i]._position;
519  if (a.length() > particle._vortexRange)
520  {
521  continue;
522  }
523 
524  b = a % particle._normal;
525  b.normalize();
526  b *= (particle._vortexRange - a.length()) / particle._vortexRange;
527  b *= particle._torque * torqueFactor;
528  a.normalize();
529  a *= _vorticityConfinement;
530  b += a;
531 
532  _particles[i]._velocity += b * 0.001f; // TODO 0.001 ???
533  }
534  }
535 
536  particle._position += particle._velocity;
537 
538  particleIter++;
539  index++;
540  }
541  }
542

544  _emissionRateGradient.getValue(particleSystemTime / __IGOR_SECOND__, emissionRate);
545  _emissionImpulseStack += emissionRate;
546  int32 createCount = static_cast<int32>(_emissionImpulseStack);
547  _emissionImpulseStack -= static_cast<float32>(createCount);
548  createParticles(createCount, emitter, particleSystemTime / __IGOR_SECOND__);
549 
550  _playbackTime += 1000.0 / _simulationRate;

552  }
553  }
554 
555  // TODO this seems a bit inefficient
556  if (!_particles.empty())
557  {
558  iaVector3f minPos = _particles.front()._position;
559  iaVector3f maxPos = _particles.front()._position;
560

565  minPos._x = _particles[i]._position._x;
566  }
567

581  }
582
