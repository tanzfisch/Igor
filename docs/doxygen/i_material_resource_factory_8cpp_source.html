199  void iMaterialResourceFactory::destroyMaterial(uint64 materialID)
200  {
201  _mutexMaterial.lock();
202 
203  auto iterMaterial = _materials.find(materialID);
204  if (iterMaterial != _materials.end())
205  {
206  (*iterMaterial).second->_isValid = false;
207 
208  auto sortedIter = find(_sortedMaterials.begin(), _sortedMaterials.end(), (*iterMaterial).second);
209 
210  con_assert_sticky(sortedIter != _sortedMaterials.end(), "inconsistent material list");
211 
212  if (sortedIter != _sortedMaterials.end())
213  {
214  _sortedMaterials.erase(sortedIter);
215  }
216 
217  _materials.erase(iterMaterial);
218  }
219  else
220  {
221  con_err("material id: " << materialID << " does not exist");
222  }
223 
224  _mutexMaterial.unlock();
225 
226  _materialDestroyedEvent(materialID);
