33  os_event._uMsg = uMsg;
34  os_event._wParam = wParam;

36

38 
39  if (WM_CREATE != uMsg)
40  {
41  window = (iWindow*)GetWindowLongPtr(_hWnd, GWLP_USERDATA);

63  }

68  window->_hWnd = nullptr;
